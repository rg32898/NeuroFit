3. /apps/api/prisma/schema.prisma with these models (use snake
   case in DB via @map, camelCase in code):

     generator client       — provider "prisma-client-js"
     datasource db          — provider "postgresql", url env("DATABASE_URL")

     model User
       id              String    @id @default(cuid())
       email           String?   @unique
       passwordHash    String?
       appleSub        String?   @unique
       googleSub       String?   @unique
       createdAt       DateTime  @default(now())
       deletedAt       DateTime?
       profile         Profile?
       progress        ProgressEvent[]
       sessions        WorkoutSession[]
       subscription    Subscription?
       reports         ContentReport[]

     model Profile
       userId          String    @id
       user            User      @relation(fields: [userId], references: [id], onDelete: Cascade)
       displayName     String?
       birthYear       Int?
       focusDomain     String?    // "vocabulary" | "writing" | ...
       relaxedMode     Boolean    @default(true)
       timerScale      Int        @default(100)  // 100|125|150|175|200 = 100% / 75% / 50% / 25% / off
       createdAt       DateTime   @default(now())
       updatedAt       DateTime   @updatedAt

     model ProficiencyScore
       id              String    @id @default(cuid())
       userId          String
       domain          String     // "vocabulary","writing","reading","speaking","math","memory"
       score           Int        // 0..5000
       updatedAt       DateTime   @updatedAt
       @@unique([userId, domain])

     model Game
       id              String    @id @default(cuid())
       slug            String    @unique
       title           String
       domain          String
       description     String
       averageDurationSec Int
       supportsRelaxed Boolean    @default(true)
       isFreeTier      Boolean    @default(false)
       isPublished     Boolean    @default(false)
       createdAt       DateTime   @default(now())
       items           GameItem[]

     model GameItem
       id              String    @id @default(cuid())
       gameId          String
       game            Game       @relation(fields: [gameId], references: [id], onDelete: Cascade)
       payload         Json       // game-specific shape
       difficultyBand  Int        // 1..5
       version         Int        @default(1)
       reviewedById    String?
       reviewedAt      DateTime?
       isPublished     Boolean    @default(false)
       createdAt       DateTime   @default(now())
       reports         ContentReport[]

     model WorkoutSession
       id              String    @id @default(cuid())
       userId          String
       user            User       @relation(fields: [userId], references: [id], onDelete: Cascade)
       date            DateTime   @default(now())
       gamesPlanned    Json       // array of game ids
       completedAt     DateTime?
       events          ProgressEvent[]

     model ProgressEvent
       id              String    @id @default(cuid())
       userId          String
       user            User       @relation(fields: [userId], references: [id], onDelete: Cascade)
       sessionId       String?
       session         WorkoutSession? @relation(fields: [sessionId], references: [id])
       eventType       String     // "GAME_COMPLETED" | "STREAK_FREEZE_USED" | ...
       gameId          String?
       itemId          String?
       score           Int?
       durationMs      Int?
       payload         Json?
       clientEventId   String     // device-generated UUID for idempotency
       createdAt       DateTime   @default(now())
       @@unique([userId, clientEventId])

     model Streak
       userId          String    @id
       user            User       @relation(fields: [userId], references: [id], onDelete: Cascade)
       current         Int        @default(0)
       longest         Int        @default(0)
       lastActiveDate  DateTime?
       freezesAvailable Int       @default(2)
       freezesResetAt  DateTime?
       updatedAt       DateTime   @updatedAt

     model Subscription
       userId          String    @id
       user            User       @relation(fields: [userId], references: [id], onDelete: Cascade)
       status          String     // "trialing"|"active"|"cancelled"|"past_due"|"none"
       plan            String     // "free"|"premium_monthly"|"premium_annual"|"family_annual"
       currentPeriodEnd DateTime?
       cancelAtPeriodEnd Boolean  @default(false)
       provider        String     // "apple"|"google"|"stripe"
       providerSubscriptionId String?
       updatedAt       DateTime   @updatedAt

     model ContentReport
       id              String    @id @default(cuid())
       reporterId      String
       reporter        User       @relation(fields: [reporterId], references: [id])
       gameItemId      String?
       gameItem        GameItem?  @relation(fields: [gameItemId], references: [id])
       category        String     // "incorrect_answer"|"unclear"|"offensive"|"other"
       message         String
       status          String     @default("open")  // "open"|"triaged"|"resolved"|"rejected"
       createdAt       DateTime   @default(now())
